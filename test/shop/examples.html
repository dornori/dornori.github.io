Integration Guide

<!-- 1. Link styles and scripts -->
<link rel="stylesheet" href="css/shop.css">
<script src="js/config.js"></script>
<script src="js/shop.js"></script>

<!-- 2. Embed from Dornori or any site with ?lang=nl -->
<iframe src="https://lumio.shop/index.html?lang=nl"></iframe>

<!-- OR: pass lang from your own JS -->
<script>
  Promise.all([Shipping.load(), Currency.init(), Shop.loadLang()]).then(() => {
    Shop.wireLanguageSwitcher(".lumio-lang-btn");
    Shop.renderCartIcon({ target: "#my-cart-icon", fixed: false });
    Shop.renderCurrencySelector("#my-currency");
    Shop.renderProductInfo("my-product", "arc-floor-lamp");
    Shop.renderMiniCart("my-mini-cart");
    Shop.renderBackButton("#back-slot", { href: "index.html" });

    // Hero image: click button → cart, click image → quick-view modal
    Shop.attachBuyOverlay(".hero-image", "arc-floor-lamp");
  });
</script>
