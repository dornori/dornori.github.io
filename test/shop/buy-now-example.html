Change v2

No Image Click Overlay

Product card images are no longer clickable by default and no modal opens on click. The "Quick View" overlay hint has been removed entirely. To make an image navigable, set product.url — see the next section.

The attachBuyOverlay() method has been retired. Images are purely decorative unless product.url is set. 
// Before (v1) — modal opened on image click, removed in v2
Shop.attachBuyOverlay(".hero-image", "arc-floor-lamp"); // ← REMOVED

// v2 — no overlay, no modal. Image is static unless product.url is set.
// Just render the shop — images work as normal img tags.
Shop.renderShop("shop", { showFilter: true });

New Feature

Product URL (Clickable Image)

Set product.url in the JSON to make the product image a link. Leave it empty or omit it entirely and the image is non-clickable — no dead cursor, no broken behavior. 

{
  "id": "arc-floor-lamp",
  "name": "Arc Floor Lamp",
  "price": 249,

  "url": "https://lumio.shop/products/arc-floor-lamp",
  // ↑ Set this → image becomes an  link
  // Leave empty → image is a plain 
, no pointer

  "image": "images/products/arc-lamp-1.jpg",
  ...
}

When url is set the card image renders as: 

<a href="https://..." class="lumio-card-img-link">
  <img class="lumio-card-img" src="..." />
</a>

// When url is empty/absent:
<div class="lumio-card-img-wrap">
  <img class="lumio-card-img" src="..." />
</div>

New Feature

Variant Pricing

Each product can now define a variants array. Variants have their own label, price, image, weight, and stock. Any field can be omitted — it falls back to the base product value. Selecting a variant instantly updates the price, image, and stock state. 

{
  "id": "hoody",
  "name": "French Terry Hoodie",
  "price": 89,      ← base/fallback price

  "variants": [
    {
      "id": "crew-neck",
      "label": "Crew Neck",
      // no price → uses base €89
      "stock": 42
    },
    {
      "id": "hoodie",
      "label": "Hoodie",
      "price": 99,  ← overrides base
      "stock": 18
    },
    {
      "id": "zip-hoodie",
      "label": "Zip Hoodie",
      "price": 109,
      "stock": 0    ← sold out variant
    }
  ]
}

Live demo — click variants to see price and state update: 

// Helper functions (all public on Shop.*)
Shop.variantPrice(product, variantId)   // → effective EUR price
Shop.variantImage(product, variantId)   // → image src string
Shop.variantWeight(product, variantId)  // → weight in kg
Shop.variantStock(product, variantId)   // → stock count
Shop.variantInStock(product, variantId) // → boolean

// Cart — variant id is stored alongside cartKey
Shop.addToCart(product, qty, variantId);

New Feature

Related Products

Add a related array to any product JSON. Related items appear as a compact "You may also need" strip below the add-to-cart button — in both grid cards and the standalone product info widget. Each item has its own add-to-cart button. 

{
  "id": "concrete-desk-lamp",
  ...
  "related": [
    {
      "id": "lamp-cable-3m",
      "name": "Textile Cable 3m",
      "price": 18,
      "weight": 0.2,
      "image": "images/products/cable-3m.jpg",
      "description": "Braided cotton, 3 metres."
    },
    {
      "id": "dimmer-switch",
      "name": "Inline Dimmer Switch",
      "price": 24,
      "weight": 0.1,
      "image": "images/products/dimmer.jpg"
    },
    {
      "id": "replacement-bulb-e27",
      "name": "Edison Bulb E27 4W",
      "price": 12,
      "weight": 0.05,
      "image": "images/products/bulb-e27.jpg"
    }
  ]
}

Integration Guide

Complete setup snippet for any page.

<!-- 1. Styles + scripts -->
<link rel="stylesheet" href="css/shop.css">
<script src="js/config.js"></script>
<script src="js/shop.js"></script>

<!-- 2. Mount points -->
<div id="cart-icon"></div>
<div id="currency"></div>
<div id="shop"></div>

<script>
Promise.all([Shipping.load(), Currency.init(), Shop.loadLang()]).then(() => {
  Shop.wireLanguageSwitcher(".lumio-lang-btn");
  Shop.renderCartIcon({ target: "#cart-icon", fixed: false });
  Shop.renderCurrencySelector("#currency");

  // Full shop grid (filter bar included)
  Shop.renderShop("shop", { showFilter: true });

  // Or single product info widget (variants + related auto-wired)
  Shop.renderProductInfo("product-info", "concrete-desk-lamp");

  // Mini cart anywhere
  Shop.renderMiniCart("mini-cart");

  // Back button
  Shop.renderBackButton("#back-slot", { href: "index.html" });
});
</script>

/* Product JSON schema (data/products/my-product.json) */
{
  "id": "my-product",
  "name": "My Product",
  "price": 89,           ← base/fallback price
  "url": "",             ← empty = non-clickable image, set URL = link
  "image": "img/main.jpg",
  "variants": [           ← optional: per-variant label/price/image/stock
    { "id": "s", "label": "Small", "price": 79,  "stock": 20 },
    { "id": "m", "label": "Medium",              "stock": 45 },
    { "id": "l", "label": "Large",  "price": 89, "stock": 0  }
  ],
  "related": [            ← optional: accessories / add-ons shown below ATC button
    { "id": "cable", "name": "Cable", "price": 12, "weight": 0.1, "image": "..." }
  ],
  "weight": 1.5,
  "stock": 45,
  "category": "desk",
  "i18n": { "en": { "name": "...", "description": "..." } }
}
