Integration Guide

<!-- 1. Link styles and scripts -->
<link rel="stylesheet" href="css/shop.css">
<script src="js/config.js"></script>
<script src="js/shop.js"></script>

<!-- 2. Drop placeholders anywhere on your page -->
<div id="my-cart-icon"></div>
<div id="my-product"></div>
<div id="my-mini-cart"></div>

<!-- 3. Wire hero images -->
<div class="hero-image"><img src="lamp.jpg"></div>

<script>
  // Cart icon (non-fixed, inline)
  Shop.renderCartIcon({ target: "#my-cart-icon", fixed: false });

  // Product detail widget
  Shop.renderProductInfo("my-product", "arc-floor-lamp");

  // Mini cart sidebar
  Shop.renderMiniCart("my-mini-cart");

  // Buy Now overlay on any image container
  Shop.attachBuyOverlay(".hero-image", "arc-floor-lamp");
</script>
